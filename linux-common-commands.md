```bash
cp --复制
mv --移动
tar --打包
find --查找文件的位置
grep --查找文件中的内容 --color 高亮显示 -A1 -B1
touch --创建一个空文件
clear --清屏
vim 进入vim编辑器，i插入 esc 退出插入， wq保存
Ps -ef --查看进程
kill -9 强制删除进程。
chmod 修改文件权限
pwd --查看当前文件
ll --显示文件和文件的详细信息
ls --只看目录
mkdir --创建目录。
rm rf --删除文件
--help 帮助
cat 查看全部
tail -n 查看最后几行
```
